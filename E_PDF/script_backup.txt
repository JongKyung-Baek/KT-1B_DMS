create PROCEDURE "INSA" 
 IS
USER_CODE VARCHAR2(50);
USER_ID VARCHAR2(20);
USER_NAME VARCHAR2(20);
USER_PWD VARCHAR2(20);
COMPANY_CODE  VARCHAR2(20);
DEPT_CODE VARCHAR2(20);
EMAIL VARCHAR2(50);
AUTH_LEVEL VARCHAR2(20);
ZIP_C0DE VARCHAR2(20);
USER_YN VARCHAR2(20);
CREATE_USER VARCHAR2(20);
CREATE_DATE VARCHAR2(20);
UPDATE_USER  VARCHAR2(20);
UPDATE_DATE VARCHAR2(20);
 --외부사용자 
 CURSOR  OUT_USER_CUR  IS
 SELECT ID USER_ID, USERNAME USER_NAME, PASSWD USER_PWD,(SELECT COMPANY_CODE FROM ITS_COMPANY WHERE COMPANY_CODE_CUSTOM =VENDERCODE) COMPANY_CODE,'' DEPT_CODE ,E_MAIL EMAIL,2 AUTH_LEVEL
 FROM WEBVENDER@LN_TOAD_DIDST2.CORP.DOOSAN.COM
  WHERE ID NOT IN (SELECT USER_ID from ITS_USER where auth_level=2);
BEGIN 
 OPEN  OUT_USER_CUR;
   LOOP 
    FETCH OUT_USER_CUR  INTO  USER_ID,USER_NAME ,USER_PWD,COMPANY_CODE ,DEPT_CODE , EMAIL , AUTH_LEVEL ;
    EXIT WHEN OUT_USER_CUR%NOTFOUND;  
      BEGIN  
       INSERT INTO  ITS_USER (USER_CODE , USER_ID,USER_NAME ,USER_PWD,COMPANY_CODE ,DEPT_CODE , EMAIL , AUTH_LEVEL) VALUES 
       ((SELECT 'USER_'||LPAD(NVL(SUBSTR(MAX(USER_CODE),6),0)+1,10,0) FROM ITS_USER2),USER_ID,USER_NAME ,USER_PWD,COMPANY_CODE ,DEPT_CODE , EMAIL , AUTH_LEVEL)
       ;
      END;  
   END LOOP;   
 dbms_output.put_line('정상적으로 처리가 완료 되었습니다.');
END;

create FUNCTION "FN_GET_CD_NM" 
(
  IN_COMBO_CD IN VARCHAR2 
, IN_VALUE IN VARCHAR2 
, IN_LANG_TYPE IN VARCHAR2 
) RETURN VARCHAR2 IS
  v_code_name VARCHAR2(2000);
BEGIN
  /* COMBO테이블에 있는 코드의 NAME을 구한다. 2019-11-01 younjh */
  SELECT
    NVL(LANG_DESC, combo.TEXT) INTO v_code_name
  FROM DOCS_COMBO combo
    LEFT JOIN DOCS_LANG lang ON lang.LANG_CD = combo.LANG_CD AND lang.LANG_TYPE = IN_LANG_TYPE
  WHERE combo.COMBO_CD = IN_COMBO_CD
  AND VALUE = IN_VALUE;
  RETURN (v_code_name);
END FN_GET_CD_NM;

/

create FUNCTION "FN_GET_USER_FULL_NM" 
(
  IN_USER_CD IN VARCHAR2 
) RETURN VARCHAR2 IS
  v_user_full_nm VARCHAR2(2000);
BEGIN  
  SELECT
    DEPT_NM || ' ' || USER_NM || ' ' || POSITION_NM INTO v_user_full_nm
  FROM DOCS_USER docsUser
  INNER JOIN DOCS_DEPT dept
  ON docsUser.DEPT_CD = dept.DEPT_CD
  INNER JOIN DOCS_POSITION docsPosition
  ON docsUser.POSITION_CD = docsPosition.POSITION_CD    
  WHERE docsUser.USER_CD = IN_USER_CD;  
  RETURN (v_user_full_nm);
END FN_GET_USER_FULL_NM;

/

(

, IN_VALUE IN VARCHAR2 
, IN_LANG_TYPE IN VARCHAR2
